Zapytanie 3: select produkt.kategoria, AVG(produkt.cena) as 'Srednia_cena', COUNT(*) as 'Liczba_produktow' from produkt GROUP BY produkt.kategoria HAVING COUNT(*) > 2

Zapytanie 6: WITH DrogieProdukty as (
    SELECT * FROM produkt where produkt.cena > 1000
)
